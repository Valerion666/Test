// Function to calculate the average of an array of numbers function calculateAverage(numbers) { if (numbers.length === 0) { return 0; } var sum = 0; for (var i = 0; i < numbers.length; i++) { sum += numbers[i]; } return sum / numbers.length; } // Example usage var data = [10, 20, 30, 40, 50]; var average = calculateAverage(data); console.log("The average is: " + average);
